const obj = { x: 1, y: 10, z: 100 }
doumi.isObject(obj) // true
doumi.omit(obj, 'z') // { x: 1, y: 10 }
doumi.forEach(obj, (value, key) => {
    console.log(value, key)
    // 1 x
    // 10 y
    // 100 z
})

const $div = doumi.create$('div', { className: 'node', innerHTML: 'welcome' })
// <div class="node">welcome</div>
